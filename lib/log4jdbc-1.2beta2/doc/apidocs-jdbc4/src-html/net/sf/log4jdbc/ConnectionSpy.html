067      /**
068       * Contains a Mapping of connectionNumber to currently open ConnectionSpy
069       * objects.
070       */
071      private static final Map connectionTracker = new HashMap();
072    
073      /**
074       * Get a dump of how many connections are open, and which connection numbers
075       * are open.
076       *
077       * @return an open connection dump.
078       */
079      public static String getOpenConnectionsDump()
080      {
081        StringBuffer dump = new StringBuffer();
082        int size;
083        Integer[] keysArr;
084        synchronized (connectionTracker)
085        {
086          size = connectionTracker.size();
087          if (size==0)
088          {
089            return "open connections:  none";
090          }
091          Set keys = connectionTracker.keySet();
092          keysArr = (Integer[]) keys.toArray(new Integer[keys.size()]);
093        }
094    
095        Arrays.sort(keysArr);
096    
097        dump.append("open connections:  ");
098        for (int i=0; i < keysArr.length; i++)
099        {
100          dump.append(keysArr[i]);
101          dump.append(" ");
102        }
103    
104        dump.append("(");
105        dump.append(size);
106        dump.append(")");
107        return dump.toString();
108      }

120      /**
121       * Create a new ConnectionSpy that wraps a given Connection.
122       *
123       * @param realConnection &quot;real&quot; Connection that this ConnectionSpy wraps.
124       * @param rdbmsSpecifics the RdbmsSpecifics object for formatting logging appropriate for the Rdbms used.
125       */
126      public ConnectionSpy(Connection realConnection, RdbmsSpecifics rdbmsSpecifics)
127      {
128        if (rdbmsSpecifics == null)
129        {
130          rdbmsSpecifics = DriverSpy.defaultRdbmsSpecifics;
131        }
132        setRdbmsSpecifics(rdbmsSpecifics);
133        if (realConnection == null)
134        {
135          throw new IllegalArgumentException("Must pass in a non null real Connection");
136        }
137        this.realConnection = realConnection;
138        log = SpyLogFactory.getSpyLogDelegator();
139    
140        synchronized (connectionTracker)
141        {
142          connectionNumber = new Integer(++lastConnectionNumber);
143          connectionTracker.put(connectionNumber, this);
144        }
145        log.connectionOpened(this);
146        reportReturn("new Connection");
147      }

181      protected void reportException(String methodCall, SQLException exception, String sql)
182      {
183        log.exceptionOccured(this, methodCall, exception, sql, -1L);
184      }
185    
186      protected void reportException(String methodCall, SQLException exception)
187      {
188        log.exceptionOccured(this, methodCall, exception, null, -1L);
189      }
190    
191      protected void reportAllReturns(String methodCall, String returnValue)
192      {
193        log.methodReturned(this, methodCall, returnValue);
194      }
195    
196      private boolean reportReturn(String methodCall, boolean value)
197      {
198        reportAllReturns(methodCall, "" + value);
199        return value;
200      }

203      {
