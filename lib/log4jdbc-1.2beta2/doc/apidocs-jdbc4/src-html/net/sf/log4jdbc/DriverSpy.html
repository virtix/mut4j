034    /**
035     * A JDBC driver which is a facade that delegates to one or more real underlying
036     * JDBC drivers.  The driver will spy on any other JDBC driver that is loaded,
037     * simply by prepending <code>jdbc:log4</code> to the normal jdbc driver URL
038     * used by any other JDBC driver. The driver, by default, also loads several
039     * well known drivers at class load time, so that this driver can be
040     * "dropped in" to any Java program that uses these drivers without making any
041     * code changes.
042     * <p/>
043     * The well known driver classes that are loaded are:
044     * <p/>
045     * <p/>
046     * <code>
047     * <ul>
048     * <li>oracle.jdbc.driver.OracleDriver</li>
049     * <li>com.sybase.jdbc2.jdbc.SybDriver</li>
050     * <li>net.sourceforge.jtds.jdbc.Driver</li>
051     * <li>com.microsoft.jdbc.sqlserver.SQLServerDriver</li>
052     * <li>com.microsoft.sqlserver.jdbc.SQLServerDriver</li>
053     * <li>weblogic.jdbc.sqlserver.SQLServerDriver</li>
054     * <li>com.informix.jdbc.IfxDriver</li>
055     * <li>org.apache.derby.jdbc.ClientDriver</li>
056     * <li>org.apache.derby.jdbc.EmbeddedDriver</li>
057     * <li>com.mysql.jdbc.Driver</li>
058     * <li>org.postgresql.Driver</li>
059     * <li>org.hsqldb.jdbcDriver</li>
060     * <li>org.h2.Driver</li>
061     * </ul>
062     * </code>
063     * <p/>
064     * <p/>
065     * Additional drivers can be set via a property: <b>log4jdbc.drivers</b>
066     * This can be either a single driver class name or a list of comma separated
067     * driver class names.
068     * <p/>
069     * The autoloading behavior can be disabled by setting a property:
070     * <b>log4jdbc.auto.load.popular.drivers</b> to false.  If that is done, then
071     * the only drivers that log4jdbc will attempt to load are the ones specified
072     * in <b>log4jdbc.drivers</b>.
073     * <p/>
074     * If any of the above driver classes cannot be loaded, the driver continues on
075     * without failing.
076     * <p/>
077     * Note that the <code>getMajorVersion</code>, <code>getMinorVersion</code> and
078     * <code>jdbcCompliant</code> method calls attempt to delegate to the last
079     * underlying driver requested through any other call that accepts a JDBC URL.
080     * <p/>
081     * This can cause unexpected behavior in certain circumstances.  For example,
082     * if one of these 3 methods is called before any underlying driver has been
083     * established, then they will return default values that might not be correct
084     * in all situations.  Similarly, if this spy driver is used to spy on more than
085     * one underlying driver concurrently, the values returned by these 3 method
086     * calls may change depending on what the last underlying driver used was at the
087     * time.  This will not usually be a problem, since the driver is retrieved by
088     * it's URL from the DriverManager in the first place (thus establishing an
089     * underlying real driver), and in most applications their is only one database.
090     *
091     * @author Arthur Blake
092     */

219      /**
220       * Get a Long option from a property and
221       * log a debug message about this.
222       *
223       * @param props Properties to get option from.
224       * @param propName property key.
225       *
226       * @return the value of that property key, converted
227       * to a Long.  Or null if not defined or is invalid.
228       */
229      private static Long getLongOption(Properties props, String propName)
230      {
231        String propValue = props.getProperty(propName);
232        Long longPropValue = null;
233        if (propValue == null)
234        {
235          log.debug("x " + propName + " is not defined");
236        }
237        else
238        {
239          try
240          {
241            longPropValue = new Long(Long.parseLong(propValue));
242            log.debug("  " + propName + " = " + longPropValue);
243          }
244          catch (NumberFormatException n)
245          {
246            log.debug("x " + propName + " \"" + propValue  +
247              "\" is not a valid number");
248          }
249        }
250        return longPropValue;
251      }

318       * @param props Properties to get option from.

320       * @param defaultValue default value to use if undefined.

325      private static boolean getBooleanOption(Properties props, String propName,

327      {
328        String propValue = props.getProperty(propName);
329        boolean val;
330        if (propValue == null)
331        {
332          log.debug("x " + propName + " is not defined (using default value " +

334          return defaultValue;
335        }
336        else
337        {
338          propValue = propValue.trim().toLowerCase();
339          if (propValue.length() == 0)

341            val = defaultValue;
342          }
343          else
344          {
345            val= "true".equals(propValue) ||
346              "yes".equals(propValue) || "on".equals(propValue);
347          }
348        }
349        log.debug("  " + propName + " = " + val);
350        return val;
351      }

354      {
355        log.debug("... log4jdbc initializing ...");
356    
357        InputStream propStream =
358          DriverSpy.class.getResourceAsStream("/log4jdbc.properties");
359    
360        Properties props = new Properties(System.getProperties());
361        if (propStream != null)
362        {
363          try
364          {
365            props.load(propStream);
366          }
367          catch (IOException e)
368          {
369            log.debug("ERROR!  io exception loading " +
370              "log4jdbc.properties from classpath: " + e.getMessage());
371          }
372          finally
373          {
374            try
375            {
376              propStream.close();
377            }
378            catch (IOException e)
379            {
380              log.debug("ERROR!  io exception closing property file stream: " +
381                e.getMessage());
382                                    }
383          }
384          log.debug("  log4jdbc.properties loaded from classpath");
385        }
386        else
387        {
388          log.debug("  log4jdbc.properties not found on classpath");
389        }
390    
391        // look for additional driver specified in properties
392        DebugStackPrefix = getStringOption(props, "log4jdbc.debug.stack.prefix");
393        TraceFromApplication = DebugStackPrefix != null;
394    
395        Long thresh = getLongOption(props, "log4jdbc.sqltiming.warn.threshold");
396        SqlTimingWarnThresholdEnabled = (thresh != null);
397        if (SqlTimingWarnThresholdEnabled)
398        {
399          SqlTimingWarnThresholdMsec = thresh.longValue();
400        }
401    
402        thresh = getLongOption(props, "log4jdbc.sqltiming.error.threshold");
403        SqlTimingErrorThresholdEnabled = (thresh != null);
404        if (SqlTimingErrorThresholdEnabled)
405        {
406          SqlTimingErrorThresholdMsec = thresh.longValue();
407        }
408    
409        DumpBooleanAsTrueFalse =
410          getBooleanOption(props, "log4jdbc.dump.booleanastruefalse",false);
411    
412        DumpSqlMaxLineLength = getLongOption(props,
413          "log4jdbc.dump.sql.maxlinelength", 90L).intValue();
414    
415        DumpFullDebugStackTrace =
416          getBooleanOption(props, "log4jdbc.dump.fulldebugstacktrace",false);
417    
418        StatementUsageWarn =
419          getBooleanOption(props, "log4jdbc.statement.warn",false);
420    
421        DumpSqlSelect = getBooleanOption(props, "log4jdbc.dump.sql.select",true);
422        DumpSqlInsert = getBooleanOption(props, "log4jdbc.dump.sql.insert",true);
423        DumpSqlUpdate = getBooleanOption(props, "log4jdbc.dump.sql.update",true);
424        DumpSqlDelete = getBooleanOption(props, "log4jdbc.dump.sql.delete",true);
425        DumpSqlCreate = getBooleanOption(props, "log4jdbc.dump.sql.create",true);
426    
427        DumpSqlFilteringOn = !(DumpSqlSelect && DumpSqlInsert && DumpSqlUpdate &&
428          DumpSqlDelete && DumpSqlCreate);
429    
430        DumpSqlAddSemicolon = getBooleanOption(props,
431          "log4jdbc.dump.sql.addsemicolon", false);
432    
433        AutoLoadPopularDrivers = getBooleanOption(props,
434          "log4jdbc.auto.load.popular.drivers", true);
435    
436        TrimSql = getBooleanOption(props, "log4jdbc.trim.sql", true);
437    
438        SuppressGetGeneratedKeysException =
439            getBooleanOption(props, "log4jdbc.suppress.generated.keys.exception",
440            false);
441    
442        // The Set of drivers that the log4jdbc driver will preload at instantiation
443        // time.  The driver can spy on any driver type, it's just a little bit
444        // easier to configure log4jdbc if it's one of these types!
445    
446        Set subDrivers = new TreeSet();
447    
448        if (AutoLoadPopularDrivers)
449        {
450          subDrivers.add("oracle.jdbc.driver.OracleDriver");
451          subDrivers.add("oracle.jdbc.OracleDriver");
452          subDrivers.add("com.sybase.jdbc2.jdbc.SybDriver");
453          subDrivers.add("net.sourceforge.jtds.jdbc.Driver");
454    
455          // MS driver for Sql Server 2000
456          subDrivers.add("com.microsoft.jdbc.sqlserver.SQLServerDriver");
457    
458          // MS driver for Sql Server 2005
459          subDrivers.add("com.microsoft.sqlserver.jdbc.SQLServerDriver");
460    
461          subDrivers.add("weblogic.jdbc.sqlserver.SQLServerDriver");
462          subDrivers.add("com.informix.jdbc.IfxDriver");
463          subDrivers.add("org.apache.derby.jdbc.ClientDriver");
464          subDrivers.add("org.apache.derby.jdbc.EmbeddedDriver");
465          subDrivers.add("com.mysql.jdbc.Driver");
466          subDrivers.add("org.postgresql.Driver");
467          subDrivers.add("org.hsqldb.jdbcDriver");
468          subDrivers.add("org.h2.Driver");
469        }
470    
471        // look for additional driver specified in properties
472        String moreDrivers = getStringOption(props, "log4jdbc.drivers");
473    
474        if (moreDrivers != null)
475        {
476          String[] moreDriversArr = moreDrivers.split(",");
477    
478          for (int i = 0; i < moreDriversArr.length; i++)
479          {
480            subDrivers.add(moreDriversArr[i]);
481            log.debug ("    will look for specific driver " + moreDriversArr[i]);
482          }
483        }
484    
485        try
486        {
487          DriverManager.registerDriver(new DriverSpy());
488        }
489        catch (SQLException s)
490        {
491          // this exception should never be thrown, JDBC just defines it
492          // for completeness
493          throw (RuntimeException) new RuntimeException
494            ("could not register log4jdbc driver!").initCause(s);
495        }
496    
497        // instantiate all the supported drivers and remove
498        // those not found
499        String driverClass;
500        for (Iterator i = subDrivers.iterator(); i.hasNext();)
501        {
502          driverClass = (String) i.next();
503          try
504          {
505            Class.forName(driverClass);
506            log.debug("  FOUND DRIVER " + driverClass);
507          }
508          catch (Throwable c)
509          {
510            i.remove();
511          }
512        }
513    
514        if (subDrivers.size() == 0)
515        {
516          log.debug("WARNING!  " +
517            "log4jdbc couldn't find any underlying jdbc drivers.");
518        }
519    
520        SqlServerRdbmsSpecifics sqlServer = new SqlServerRdbmsSpecifics();
521        OracleRdbmsSpecifics oracle = new OracleRdbmsSpecifics();
522    
523        /** create lookup Map for specific rdbms formatters */
524        rdbmsSpecifics = new HashMap();
525        rdbmsSpecifics.put("oracle.jdbc.driver.OracleDriver", oracle);
526        rdbmsSpecifics.put("oracle.jdbc.OracleDriver", oracle);
527        rdbmsSpecifics.put("net.sourceforge.jtds.jdbc.Driver", sqlServer);
528        rdbmsSpecifics.put("com.microsoft.jdbc.sqlserver.SQLServerDriver",
529          sqlServer);
530        rdbmsSpecifics.put("weblogic.jdbc.sqlserver.SQLServerDriver", sqlServer);
531    
532        log.debug("... log4jdbc initialized! ...");
533      }

535      static RdbmsSpecifics defaultRdbmsSpecifics = new RdbmsSpecifics();

537      /**
538       * Get the RdbmsSpecifics object for a given Connection.
539       *
540       * @param conn JDBC connection to get RdbmsSpecifics for.
541       * @return RdbmsSpecifics for the given connection.
542       */
543      static RdbmsSpecifics getRdbmsSpecifics(Connection conn)
544      {
545        String driverName = "";
546        try
547        {
548          DatabaseMetaData dbm = conn.getMetaData();
549          driverName = dbm.getDriverName();
550        }
551        catch (SQLException s)
552        {
553          // silently fail
554        }
555    
556        log.debug("driver name is " + driverName);
557    
558        RdbmsSpecifics r = (RdbmsSpecifics) rdbmsSpecifics.get(driverName);
559    
560        if (r == null)
561        {
562          return defaultRdbmsSpecifics;
563        }
564        else
565        {
566          return r;
567        }
568      }

629      /**
630       * Returns true if this is a <code>jdbc:log4</code> URL and if the URL is for
631       * an underlying driver that this DriverSpy can spy on.
632       *
633       * @param url JDBC URL.
634       *
635       * @return true if this Driver can handle the URL.
636       *
637       * @throws SQLException if a database access error occurs
638       */
639      public boolean acceptsURL(String url) throws SQLException
640      {
641        Driver d = getUnderlyingDriver(url);
642        if (d != null)
643        {
644          lastUnderlyingDriverRequested = d;
645          return true;
646        }
647        else
648        {
649          return false;
650        }
651      }

653      /**
654       * Given a <code>jdbc:log4</code> type URL, find the underlying real driver
655       * that accepts the URL.
656       *
657       * @param url JDBC connection URL.
658       *
659       * @return Underlying driver for the given URL. Null is returned if the URL is
660       *         not a <code>jdbc:log4</code> type URL or there is no underlying
661       *         driver that accepts the URL.
662       *
663       * @throws SQLException if a database access error occurs.
664       */
665      private Driver getUnderlyingDriver(String url) throws SQLException
666      {
667        if (url.startsWith("jdbc:log4"))
668        {
669          url = url.substring(9);
670    
671          Enumeration e = DriverManager.getDrivers();
672    
673          Driver d;
674          while (e.hasMoreElements())
675          {
676            d = (Driver) e.nextElement();
677    
678            if (d.acceptsURL(url))
679            {
680              return d;
681            }
682          }
683        }
684        return null;
685      }

687      /**
688       * Get a Connection to the database from the underlying driver that this
689       * DriverSpy is spying on.  If logging is not enabled, an actual Connection to
690       * the database returned.  If logging is enabled, a ConnectionSpy object which
691       * wraps the real Connection is returned.
692       *
693       * @param url  JDBC connection URL
694       * .
695       * @param info a list of arbitrary string tag/value pairs as
696       *             connection arguments. Normally at least a "user" and
697       *             "password" property should be included.
698       *
699       * @return     a <code>Connection</code> object that represents a
700       *             connection to the URL.
701       *
702       * @throws SQLException if a database access error occurs
703       */
704      public Connection connect(String url, Properties info) throws SQLException
705      {
706        Driver d = getUnderlyingDriver(url);
707        if (d == null)
708        {
709          return null;
710        }
711    
712        // get actual URL that the real driver expects
713        // (strip off "jdbc:log4" from url)
714        url = url.substring(9);
715    
716        lastUnderlyingDriverRequested = d;
717        Connection c = d.connect(url, info);
718    
719        if (c == null)
720        {
721          throw new SQLException("invalid or unknown driver url: " + url);
722        }
723        if (log.isJdbcLoggingEnabled())
724        {
725          ConnectionSpy cspy = new ConnectionSpy(c);
726          RdbmsSpecifics r = null;
727          String dclass = d.getClass().getName();
728          if (dclass != null && dclass.length() > 0)
729          {
730            r = (RdbmsSpecifics) rdbmsSpecifics.get(dclass);
731          }
732    
733          if (r == null)
734          {
735            r = defaultRdbmsSpecifics;
736          }
737          cspy.setRdbmsSpecifics(r);
738          return cspy;
739        }
740        else
741        {
742          return c;
743        }
744      }

746      /**
747       * Gets information about the possible properties for the underlying driver.
748       *
749       * @param url  the URL of the database to which to connect
750       *
751       * @param info a proposed list of tag/value pairs that will be sent on
752       *             connect open
753       * @return     an array of <code>DriverPropertyInfo</code> objects describing
754       *             possible properties.  This array may be an empty array if no
755       *             properties are required.
756       *
757       * @throws SQLException if a database access error occurs
758       */
759      public DriverPropertyInfo[] getPropertyInfo(String url, Properties info)
760        throws SQLException
761      {
762        Driver d = getUnderlyingDriver(url);
763        if (d == null)
764        {
765          return new DriverPropertyInfo[0];
766        }
767    
768        lastUnderlyingDriverRequested = d;
769        return d.getPropertyInfo(url, info);
770      }
