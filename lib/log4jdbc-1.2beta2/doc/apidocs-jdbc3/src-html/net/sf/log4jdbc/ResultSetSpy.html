037    /**
038     * Wraps a ResultSet and reports method calls, returns and exceptions.
039     *
040     * JDBC 3 version.
041     *
042     * @author Arthur Blake
043     */
044    public class ResultSetSpy implements ResultSet, Spy
045    {
046      private final SpyLogDelegator log;
047    
048      /**
049       * Report an exception to be logged.
050       *
051       * @param methodCall description of method call and arguments passed to it that generated the exception.
052       * @param exception exception that was generated
053       */
054      protected void reportException(String methodCall, SQLException exception)
055      {
056        log.exceptionOccured(this, methodCall, exception, null, -1L);
057      }
058    
059      /**
060       * Report (for logging) that a method returned.  All the other reportReturn methods are conveniance methods that call
061       * this method.
062       *
063       * @param methodCall description of method call and arguments passed to it that returned.
064       * @param msg description of what the return value that was returned.  may be an empty String for void return types.
065       */
066      protected void reportAllReturns(String methodCall, String msg)
067      {
068        log.methodReturned(this, methodCall, msg);
069      }

071      private ResultSet realResultSet;

074       * Get the real ResultSet that this ResultSetSpy wraps.
075       *
076       * @return the real ResultSet that this ResultSetSpy wraps.

078      public ResultSet getRealResultSet()
079      {
080        return realResultSet;
081      }

083      private StatementSpy parent;
084    
085      /**
086       * Create a new ResultSetSpy that wraps another ResultSet object, that logs all method calls, expceptions, etc.
087       *
088       * @param parent Statement that generated this ResultSet.
089       * @param realResultSet real underlying ResultSet that is being wrapped.
090       */
091      public ResultSetSpy(StatementSpy parent, ResultSet realResultSet)
092      {
093        if (realResultSet == null)
094        {
095          throw new IllegalArgumentException("Must provide a non null real ResultSet");
096        }
097        this.realResultSet = realResultSet;
098        this.parent = parent;
099        log = SpyLogFactory.getSpyLogDelegator();
100        reportReturn("new ResultSet");
101      }
