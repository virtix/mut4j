064      /**
065       * Get a dump of how many connections are open, and which connection numbers
066       * are open.
067       *
068       * @return an open connection dump.
069       */
070      public static String getOpenConnectionsDump()
071      {
072        StringBuffer dump = new StringBuffer();
073        int size;
074        Integer[] keysArr;
075        synchronized (connectionTracker)
076        {
077          size = connectionTracker.size();
078          if (size==0)
079          {
080            return "open connections:  none";
081          }
082          Set keys = connectionTracker.keySet();
083          keysArr = (Integer[]) keys.toArray(new Integer[keys.size()]);
084        }
085    
086        Arrays.sort(keysArr);
087    
088        dump.append("open connections:  ");
089        for (int i=0; i < keysArr.length; i++)
090        {
091          dump.append(keysArr[i]);
092          dump.append(" ");
093        }
094    
095        dump.append("(");
096        dump.append(size);
097        dump.append(")");
098        return dump.toString();
099      }

111      /**
112       * Create a new ConnectionSpy that wraps a given Connection.
113       *
114       * @param realConnection "real" Connection that this ConnectionSpy wraps.
115       * @param rdbmsSpecifics the RdbmsSpecifics object for formatting logging appropriate for the Rdbms used.
116       */
117      public ConnectionSpy(Connection realConnection, RdbmsSpecifics rdbmsSpecifics)
118      {
119        if (rdbmsSpecifics == null)
120        {
121          rdbmsSpecifics = DriverSpy.defaultRdbmsSpecifics;
122        }
123        setRdbmsSpecifics(rdbmsSpecifics);
124        if (realConnection == null)
125        {
126          throw new IllegalArgumentException("Must pass in a non null real Connection");
127        }
128        this.realConnection = realConnection;
129        log = SpyLogFactory.getSpyLogDelegator();
130    
131        synchronized (connectionTracker)
132        {
133          connectionNumber = new Integer(++lastConnectionNumber);
134          connectionTracker.put(connectionNumber, this);
135        }
136        log.connectionOpened(this);
137        reportReturn("new Connection");
138      }

172      protected void reportException(String methodCall, SQLException exception, String sql)
173      {
174        log.exceptionOccured(this, methodCall, exception, sql, -1L);
175      }
176    
177      protected void reportException(String methodCall, SQLException exception)
178      {
179        log.exceptionOccured(this, methodCall, exception, null, -1L);
180      }
181    
182      protected void reportAllReturns(String methodCall, String returnValue)
183      {
184        log.methodReturned(this, methodCall, returnValue);
185      }
186    
187      private boolean reportReturn(String methodCall, boolean value)
188      {
189        reportAllReturns(methodCall, "" + value);
190        return value;
191      }
192    
193      private int reportReturn(String methodCall, int value)
194      {
195        reportAllReturns(methodCall, "" + value);
196        return value;
197      }
198    
199      private Object reportReturn(String methodCall, Object value)
200      {
201        reportAllReturns(methodCall, "" + value);
202        return value;
203      }
204    
205      private void reportReturn(String methodCall)
206      {
207        reportAllReturns(methodCall, "");
208      }

731      public void close() throws SQLException
732      {
733        String methodCall = "close()";
734        try
735        {
736          realConnection.close();
737        }
738        catch (SQLException s)
739        {
740          reportException(methodCall, s);
741          throw s;
742        }
743        finally
744        {
745          synchronized (connectionTracker)
746          {
747            connectionTracker.remove(connectionNumber);
748          }
749          log.connectionClosed(this);
750        }
751        reportReturn(methodCall);
752      }
