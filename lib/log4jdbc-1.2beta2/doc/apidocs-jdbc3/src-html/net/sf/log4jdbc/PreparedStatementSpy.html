051      // a way to turn on and off type help...
052      // todo:  make this a configurable parameter
053      // todo, debug arrays and streams in a more useful manner.... if possible
054      private static final boolean showTypeHelp = false;
055    
056      /**
057       * Store an argument (bind variable) into the argTrace list (above) for later dumping.
058       *
059       * @param i          index of argument being set.
060       * @param typeHelper optional additional info about the type that is being set in the arg
061       * @param arg        argument being bound.
062       */
063      protected void argTraceSet(int i, String typeHelper, Object arg)
064      {
065        String tracedArg;
066        try
067        {
068          tracedArg = rdbmsSpecifics.formatParameterObject(arg);
069        }
070        catch (Throwable t)
071        {
072          // rdbmsSpecifics should NEVER EVER throw an exception!!
073          // but just in case it does, we trap it.
074          log.debug("rdbmsSpecifics threw an exception while trying to format a " +
075            "parameter object [" + arg + "] this is very bad!!! (" +
076            t.getMessage() + ")");
077    
078          // backup - so that at least we won't harm the application using us
079          tracedArg = arg==null?"null":arg.toString();
080        }
081    
082        i--;  // make the index 0 based
083        synchronized (argTrace)
084        {
085          // if an object is being inserted out of sequence, fill up missing values with null...
086          while (i >= argTrace.size())
087          {
088            argTrace.add(argTrace.size(), null);
089          }
090          if (!showTypeHelp || typeHelper == null)
091          {
092            argTrace.set(i, tracedArg);
093          }
094          else
095          {
096            argTrace.set(i, typeHelper + tracedArg);
097          }
098        }
099      }

101      private String sql;
102    
103      protected String dumpedSql()
104      {
105        StringBuffer dumpSql = new StringBuffer();
106        int lastPos = 0;
107        int Qpos = sql.indexOf('?', lastPos);  // find position of first question mark
108        int argIdx = 0;
109        String arg;
110    
111        while (Qpos != -1)
112        {
113          // get stored argument
114          synchronized (argTrace)
115          {
116            try
117            {
118              arg = (String) argTrace.get(argIdx);
119            }
120            catch (IndexOutOfBoundsException e)
121            {
122              arg = "?";
123            }
124          }
125          if (arg == null)
126          {
127            arg = "?";
128          }
129    
130          argIdx++;
131    
132          dumpSql.append(sql.substring(lastPos, Qpos));  // dump segment of sql up to question mark.
133          lastPos = Qpos + 1;
134          Qpos = sql.indexOf('?', lastPos);
135          dumpSql.append(arg);
136        }
137        if (lastPos < sql.length())
138        {
139          dumpSql.append(sql.substring(lastPos, sql.length()));  // dump last segment
140        }
141    
142        return dumpSql.toString();
143      }

170      /**
171       * Create a PreparedStatementSpy (JDBC 3 version) for logging activity of another PreparedStatement.
172       *
173       * @param sql                   SQL for the prepared statement that is being spied upon.
174       * @param connectionSpy         ConnectionSpy that was called to produce this PreparedStatement.
175       * @param realPreparedStatement The actual PreparedStatement that is being spied upon.
176       */
177      public PreparedStatementSpy(String sql, ConnectionSpy connectionSpy, PreparedStatement realPreparedStatement)
178      {
179        super(connectionSpy, realPreparedStatement);  // does null check for us
180        this.sql = sql;
181        this.realPreparedStatement = realPreparedStatement;
182        rdbmsSpecifics = connectionSpy.getRdbmsSpecifics();
183      }

405      public boolean execute() throws SQLException
406      {
407        String methodCall = "execute()";
408        String dumpedSql = dumpedSql();
409        reportSql(dumpedSql, methodCall);
410        long tstart = System.currentTimeMillis();
411        try
412        {
413          boolean result = realPreparedStatement.execute();
414          reportSqlTiming(System.currentTimeMillis() - tstart, dumpedSql, methodCall);
415          return reportReturn(methodCall, result);
416        }
417        catch (SQLException s)
418        {
419          reportException(methodCall, s, dumpedSql, System.currentTimeMillis() - tstart);
420          throw s;
421        }
422      }
